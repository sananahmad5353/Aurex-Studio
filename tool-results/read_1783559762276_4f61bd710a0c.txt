    33→interface AdminPanelProps { open: boolean; onClose: () => void; }

    36→
    37→export default function AdminPanel({ open, onClose }: AdminPanelProps) {
    38→  const [token, setToken] = useState<string | null>(null);
    39→  const [email, setEmail] = useState('');
    40→  const [password, setPassword] = useState('');
    41→  const [loginLoading, setLoginLoading] = useState(false);
    42→  const [loginError, setLoginError] = useState('');
    43→  const [activeTab, setActiveTab] = useState<Tab>('settings');
    44→  const [loading, setLoading] = useState(false);
    45→  const { toast } = useToast();
    46→
    47→  const [settings, setSettings] = useState<Record<string, string>>({});
    48→  const [aboutStats, setAboutStats] = useState<{ label: string; value: string }[]>([]);
    49→  const [slides, setSlides] = useState<Slide[]>([]);
    50→  const [editingSlide, setEditingSlide] = useState<Partial<Slide> | null>(null);
    51→  const [slideFormOpen, setSlideFormOpen] = useState(false);
    52→  const [services, setServices] = useState<ServiceItem[]>([]);
    53→  const [editingService, setEditingService] = useState<Partial<ServiceItem> | null>(null);
    54→  const [serviceFormOpen, setServiceFormOpen] = useState(false);
    55→  const [testimonials, setTestimonials] = useState<TestimonialItem[]>([]);
    56→  const [editingTestimonial, setEditingTestimonial] = useState<Partial<TestimonialItem> | null>(null);
    57→  const [testimonialFormOpen, setTestimonialFormOpen] = useState(false);
    58→  const [partners, setPartners] = useState<PartnerItem[]>([]);
    59→  const [editingPartner, setEditingPartner] = useState<Partial<PartnerItem> | null>(null);
    60→  const [partnerFormOpen, setPartnerFormOpen] = useState(false);
    61→  const [messages, setMessages] = useState<ContactMsg[]>([]);
    62→  const [reels, setReels] = useState<ReelItem[]>([]);
    63→  const [editingReel, setEditingReel] = useState<Partial<ReelItem> | null>(null);
    64→  const [reelFormOpen, setReelFormOpen] = useState(false);
    65→  const [aboutImages, setAboutImages] = useState<string[]>([]);
    66→  const [currentPwd, setCurrentPwd] = useState('');
    67→  const [newPwd, setNewPwd] = useState('');
    68→  const [confirmPwd, setConfirmPwd] = useState('');
    69→  // 2FA states
    70→  const [twoFactorRequired, setTwoFactorRequired] = useState(false);
    71→  const [pendingToken, setPendingToken] = useState<string | null>(null);
    72→  const [twoFactorCode, setTwoFactorCode] = useState('');
    73→  const [twoFactorEnabled, setTwoFactorEnabled] = useState(false);
    74→  const [twoFactorSecret, setTwoFactorSecret] = useState('');
    75→  const [twoFactorUri, setTwoFactorUri] = useState('');
    76→  const [twoFactorSetupCode, setTwoFactorSetupCode] = useState('');
    77→  const [twoFactorVerifyCode, setTwoFactorVerifyCode] = useState('');
    78→  const [twoFactorDisableCode, setTwoFactorDisableCode] = useState('');
    79→  const [showTwoFactorSetup, setShowTwoFactorSetup] = useState(false);
    80→
    81→  const h = () => ({ 'Content-Type': 'application/json' as const, Authorization: `Bearer ${token}` as const });
    82→
    83→  const fetchData = async () => {
    84→    const hdrs = h();
    85→    try { const r = await fetch('/api/settings', { headers: hdrs }); const d = await r.json(); setSettings(d); try { setAboutStats(JSON.parse(d.aboutStats || '[]')); } catch { setAboutStats([]); } } catch { /* */ }
    86→    try { const r = await fetch('/api/hero-slides', { headers: hdrs }); if (r.ok) setSlides(await r.json()); } catch { /* */ }
    87→    try { const r = await fetch('/api/services', { headers: hdrs }); if (r.ok) setServices(await r.json()); } catch { /* */ }
    88→    try { const r = await fetch('/api/testimonials', { headers: hdrs }); if (r.ok) setTestimonials(await r.json()); } catch { /* */ }
    89→    try { const r = await fetch('/api/partners', { headers: hdrs }); if (r.ok) setPartners(await r.json()); } catch { /* */ }
    90→    try { const r = await fetch('/api/contact', { headers: hdrs }); if (r.ok) setMessages(await r.json()); } catch { /* */ }
    91→    try { setAboutImages(JSON.parse(settings.aboutImages || '[]')); } catch { setAboutImages([]); }
    92→  };
    93→
    94→  const handleLogin = async () => {
    95→    setLoginLoading(true); setLoginError('');
    96→    try {
    97→      const res = await fetch('/api/auth/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ email, password, twoFactorCode: twoFactorRequired ? twoFactorCode : undefined, pendingToken: twoFactorRequired ? pendingToken : undefined }) });
    98→      const data = await res.json();
    99→      if (res.ok) {
   100→        if (data.twoFactorRequired) {
   101→          setTwoFactorRequired(true);
   102→          setPendingToken(data.pendingToken);
   103→          setLoginError('');
   104→        } else {
   105→          setToken(data.token);
   106→          setTwoFactorRequired(false);
   107→          setPendingToken(null);
   108→          setTwoFactorCode('');
   109→          toast({ title: 'Logged in successfully' });
   110→          void fetchTwoFactorStatus();
   111→          void fetchData();
   112→        }
   113→      }
   114→      else { setLoginError(data.error || 'Login failed'); }
   115→    } catch { setLoginError('Network error'); }
   116→    setLoginLoading(false);
   117→  };
   118→
   119→  const fetchTwoFactorStatus = async () => {
   120→    if (!token) return;
   121→    try {
   122→      const r = await fetch('/api/auth/2fa', { headers: h() });
   123→      if (r.ok) { const d = await r.json(); setTwoFactorEnabled(d.enabled); }
   124→    } catch { /* */ }
   125→  };
   126→
   127→  const handleSetup2FA = async () => {
   128→    setLoading(true);
   129→    try {
   130→      const r = await fetch('/api/auth/2fa', { method: 'POST', headers: h() });
   131→      const d = await r.json();
   132→      if (r.ok) {
   133→        setTwoFactorSecret(d.secret);
   134→        setTwoFactorUri(d.uri);
   135→        setTwoFactorSetupCode(d.testCode);
   136→        setShowTwoFactorSetup(true);
   137→      } else { toast({ title: 'Error', description: d.error, variant: 'destructive' }); }
   138→    } catch { toast({ title: 'Error', variant: 'destructive' }); }
   139→    setLoading(false);
   140→  };
   141→
   142→  const handleVerify2FA = async (action: 'enable' | 'disable') => {
   143→    const code = action === 'enable' ? twoFactorVerifyCode : twoFactorDisableCode;
   144→    if (!code) { toast({ title: 'Enter the 6-digit code', variant: 'destructive' }); return; }
   145→    setLoading(true);
   146→    try {
   147→      const r = await fetch('/api/auth/2fa', { method: 'PUT', headers: h(), body: JSON.stringify({ code, action }) });
   148→      const d = await r.json();
   149→      if (r.ok) {
   150→        toast({ title: d.message });
   151→        setTwoFactorEnabled(action === 'enable');
   152→        setShowTwoFactorSetup(false);
   153→        setTwoFactorSecret('');
   154→        setTwoFactorUri('');
   155→        setTwoFactorSetupCode('');
   156→        setTwoFactorVerifyCode('');
   157→        setTwoFactorDisableCode('');
   158→        if (d.token) setToken(d.token);
   159→      } else { toast({ title: 'Error', description: d.error, variant: 'destructive' }); }
   160→    } catch { toast({ title: 'Error', variant: 'destructive' }); }
   161→    setLoading(false);
   162→  };
   163→
   164→  const handleSaveSettings = async () => {
   165→    setLoading(true);
   166→    const res = await fetch('/api/settings', { method: 'PUT', headers: h(), body: JSON.stringify({ ...settings, aboutStats: JSON.stringify(aboutStats), aboutImages: JSON.stringify(aboutImages) }) });
   167→    if (res.ok) toast({ title: 'Settings saved' }); else toast({ title: 'Error', variant: 'destructive' });
   168→    setLoading(false);
   169→  };
   170→
   171→  const handleSaveReel = async () => {
   172→    if (!editingReel?.reelUrl) { toast({ title: 'Reel URL is required', variant: 'destructive' }); return; }
   173→    setLoading(true);
   174→    if (editingReel.id) await fetch('/api/reels', { method: 'PUT', headers: h(), body: JSON.stringify(editingReel) });
   175→    else await fetch('/api/reels', { method: 'POST', headers: h(), body: JSON.stringify(editingReel) });
   176→    toast({ title: editingReel.id ? 'Reel updated' : 'Reel added' });
   177→    setEditingReel(null); setReelFormOpen(false); void fetchData(); setLoading(false);
   178→  };
   179→
   180→  const handleDeleteReel = async (id: string) => { if (!confirm('Delete?')) return; await fetch(`/api/reels?id=${id}`, { method: 'DELETE', headers: h() }); toast({ title: 'Deleted' }); void fetchData(); };
   181→
   182→  const handleSaveSlide = async () => {
   183→    if (!editingSlide?.title || !editingSlide?.subtitle || !editingSlide?.imageUrl) { toast({ title: 'Missing fields', variant: 'destructive' }); return; }
   184→    setLoading(true);
   185→    if (editingSlide.id) await fetch('/api/hero-slides', { method: 'PUT', headers: h(), body: JSON.stringify(editingSlide) });
   186→    else await fetch('/api/hero-slides', { method: 'POST', headers: h(), body: JSON.stringify(editingSlide) });
   187→    toast({ title: editingSlide.id ? 'Slide updated' : 'Slide created' });
   188→    setEditingSlide(null); setSlideFormOpen(false); void fetchData(); setLoading(false);
   189→  };
   190→
   191→  const handleDeleteSlide = async (id: string) => { if (!confirm('Delete this slide?')) return; await fetch(`/api/hero-slides?id=${id}`, { method: 'DELETE', headers: h() }); toast({ title: 'Deleted' }); void fetchData(); };
   192→
   193→  const handleSaveService = async () => {
   194→    if (!editingService?.title || !editingService?.description) { toast({ title: 'Missing fields', variant: 'destructive' }); return; }
   195→    setLoading(true);
   196→    if (editingService.id) await fetch('/api/services', { method: 'PUT', headers: h(), body: JSON.stringify(editingService) });
   197→    else await fetch('/api/services', { method: 'POST', headers: h(), body: JSON.stringify(editingService) });
   198→    toast({ title: editingService.id ? 'Service updated' : 'Service created' });
   199→    setEditingService(null); setServiceFormOpen(false); void fetchData(); setLoading(false);
   200→  };
   201→
   202→  const handleDeleteService = async (id: string) => { if (!confirm('Delete this service?')) return; await fetch(`/api/services?id=${id}`, { method: 'DELETE', headers: h() }); toast({ title: 'Deleted' }); void fetchData(); };
   203→
   204→  const handleSaveTestimonial = async () => {
   205→    if (!editingTestimonial?.name || !editingTestimonial?.content) { toast({ title: 'Missing fields', variant: 'destructive' }); return; }
   206→    setLoading(true);
   207→    const headers = h();
   208→    if (editingTestimonial.id) await fetch('/api/testimonials', { method: 'PUT', headers, body: JSON.stringify(editingTestimonial) });
   209→    else await fetch('/api/testimonials', { method: 'POST', headers, body: JSON.stringify(editingTestimonial) });
   210→    toast({ title: editingTestimonial.id ? 'Review updated' : 'Review created' });
   211→    setEditingTestimonial(null); setTestimonialFormOpen(false); void fetchData(); setLoading(false);
   212→  };
   213→
   214→  const handleDeleteTestimonial = async (id: string) => { if (!confirm('Delete this review?')) return; await fetch(`/api/testimonials?id=${id}`, { method: 'DELETE', headers: h() }); toast({ title: 'Deleted' }); void fetchData(); };
   215→
   216→  const handleSavePartner = async () => {
   217→    if (!editingPartner?.name || !editingPartner?.imageUrl) { toast({ title: 'Missing fields', variant: 'destructive' }); return; }
   218→    setLoading(true);
   219→    const headers = h();
   220→    if (editingPartner.id) await fetch('/api/partners', { method: 'PUT', headers, body: JSON.stringify(editingPartner) });
   221→    else await fetch('/api/partners', { method: 'POST', headers, body: JSON.stringify(editingPartner) });
   222→    toast({ title: editingPartner.id ? 'Partner updated' : 'Partner created' });
   223→    setEditingPartner(null); setPartnerFormOpen(false); void fetchData(); setLoading(false);
   224→  };
   225→
   226→  const handleDeletePartner = async (id: string) => { if (!confirm('Delete this partner?')) return; await fetch(`/api/partners?id=${id}`, { method: 'DELETE', headers: h() }); toast({ title: 'Deleted' }); void fetchData(); };
   227→
   228→  const handleMarkRead = async (id: string) => {
   229→    await fetch(`/api/contact?markRead=${id}`, { method: 'GET', headers: h() });
   230→    void fetchData();
   231→  };
   232→
   233→  const handleDeleteMessage = async (id: string) => { if (!confirm('Delete this message?')) return; await fetch(`/api/contact?id=${id}`, { method: 'DELETE', headers: h() }); toast({ title: 'Deleted' }); void fetchData(); };
   234→
   235→  const handleChangePassword = async () => {
   236→    if (newPwd !== confirmPwd) { toast({ title: 'Passwords do not match', variant: 'destructive' }); return; }
   237→    if (newPwd.length < 8) { toast({ title: 'Min 8 characters, with uppercase, lowercase, number & special char', variant: 'destructive' }); return; }
   238→    setLoading(true);
   239→    const res = await fetch('/api/auth/change-password', { method: 'POST', headers: h(), body: JSON.stringify({ currentPassword: currentPwd, newPassword: newPwd }) });
   240→    const d = await res.json();
   241→    if (res.ok) { toast({ title: 'Password changed' }); setCurrentPwd(''); setNewPwd(''); setConfirmPwd(''); }
   242→    else toast({ title: 'Error', description: d.error, variant: 'destructive' });
   243→    setLoading(false);
   244→  };
   245→
   246→  if (!open) return null;
   247→
   248→  // LOGIN
   249→  if (!token) {
   250→    return (
   251→      <div className="fixed inset-0 z-[100] bg-black/60 backdrop-blur-sm flex items-center justify-center p-4">
   252→        <div className="bg-white rounded-2xl shadow-2xl w-full max-w-md p-8">
   253→          <div className="flex justify-between items-center mb-6">
   254→            <h2 className="text-2xl font-bold text-slate-900">Admin Login</h2>
   255→            <button onClick={onClose} className="p-2 hover:bg-slate-100 rounded-lg transition-colors"><X size={20} className="text-slate-500" /></button>
   256→          </div>
   257→          {loginError && <div className="mb-4 p-3 bg-red-50 border border-red-200 rounded-lg text-sm text-red-600">{loginError}</div>}
   258→          {twoFactorRequired && (
   259→            <div className="mb-4 p-3 bg-amber-50 border border-amber-200 rounded-lg text-sm text-amber-700 flex items-center gap-2">
   260→              <SmartphoneNfc size={16} className="flex-shrink-0" />
   261→              Two-factor authentication required. Enter the code from your authenticator app.
   262→            </div>
   263→          )}
   264→          <div className="space-y-4">
   265→            {!twoFactorRequired && (
   266→              <>
   267→                <div><Label htmlFor="email">Email</Label><Input id="email" type="email" placeholder="Enter your email" value={email} onChange={(e) => setEmail(e.target.value)} className="mt-1.5" /></div>
   268→                <div><Label htmlFor="password">Password</Label><Input id="password" type="password" placeholder="Enter your password" value={password} onChange={(e) => setPassword(e.target.value)} onKeyDown={(e) => e.key === 'Enter' && handleLogin()} className="mt-1.5" /></div>
   269→              </>
   270→            )}
   271→            {twoFactorRequired && (
   272→              <div>
   273→                <Label htmlFor="twofa">Authentication Code</Label>
   274→                <Input id="twofa" type="text" placeholder="6-digit code" value={twoFactorCode} onChange={(e) => setTwoFactorCode(e.target.value.replace(/[^0-9]/g, '').slice(0, 6))} onKeyDown={(e) => e.key === 'Enter' && handleLogin()} className="mt-1.5 text-center text-2xl tracking-[0.5em] font-mono" maxLength={6} autoFocus />
   275→                <button onClick={() => { setTwoFactorRequired(false); setPendingToken(null); setTwoFactorCode(''); setLoginError(''); }} className="mt-2 text-xs text-slate-500 hover:text-slate-700 underline">Back to password login</button>
   276→              </div>
   277→            )}
   278→            <Button onClick={handleLogin} disabled={loginLoading || (!twoFactorRequired && (!email || !password)) || (twoFactorRequired && twoFactorCode.length !== 6)} className="w-full bg-emerald-500 hover:bg-emerald-600 text-white rounded-xl h-11">
   279→              {loginLoading && <Loader2 className="mr-2 h-4 w-4 animate-spin" />}{twoFactorRequired ? 'Verify Code' : 'Sign In'}
   280→            </Button>
   281→          </div>
   282→        </div>
   283→      </div>
   284→    );
   285→  }
   286→
   287→  // DASHBOARD
   288→  const tabs: { key: Tab; label: string; icon: React.ReactNode }[] = [
   289→    { key: 'settings', label: 'Settings', icon: <Settings size={18} /> },
   290→    { key: 'slides', label: 'Slides', icon: <ImageIcon size={18} /> },
   291→    { key: 'services', label: 'Services', icon: <Layers size={18} /> },
   292→    { key: 'reviews', label: 'Reviews', icon: <Star size={18} /> },
   293→    { key: 'partners', label: 'Partners', icon: <Shield size={18} /> },
   294→    { key: 'reels', label: 'Reels', icon: <Film size={18} /> },
   295→    { key: 'messages', label: 'Messages', icon: <MessageSquare size={18} /> },
   296→    { key: 'password', label: 'Password', icon: <KeyRound size={18} /> },
   297→    { key: 'security', label: 'Security', icon: <Shield size={18} /> },
   298→  ];
   299→
   300→  const unreadCount = messages.filter(m => !m.read).length;
   301→
   302→  return (
   303→    <div className="fixed inset-0 z-[100] bg-black/60 backdrop-blur-sm flex items-center justify-center p-2 sm:p-4">
   304→      <div className="bg-white rounded-2xl shadow-2xl w-full max-w-5xl max-h-[92vh] flex flex-col overflow-hidden">
   305→        {/* Header */}
   306→        <div className="flex items-center justify-between px-6 py-4 border-b">
   307→          <div className="flex items-center gap-3">
   308→            <div className="flex h-8 w-8 items-center justify-center rounded-lg bg-slate-900"><span className="text-sm font-bold text-white">A</span></div>
   309→            <h2 className="text-lg font-bold text-slate-900">Admin Panel</h2>
   310→          </div>
   311→          <div className="flex items-center gap-2">
   312→            <Button variant="ghost" size="sm" onClick={() => { setToken(null); setEmail(''); setPassword(''); setActiveTab('settings'); }} className="text-slate-500 hover:text-red-500"><LogOut size={16} className="mr-1.5" />Logout</Button>
   313→            <button onClick={onClose} className="p-2 hover:bg-slate-100 rounded-lg transition-colors"><X size={20} className="text-slate-500" /></button>
   314→          </div>
   315→        </div>
   316→
   317→        {/* Tabs */}
   318→        <div className="flex border-b px-4 overflow-x-auto">
   319→          {tabs.map((tab) => (
   320→            <button key={tab.key} onClick={() => { setActiveTab(tab.key); if (tab.key === 'messages') void fetchData(); }}
   321→              className={`flex items-center gap-1.5 px-3 py-3 text-sm font-medium border-b-2 transition-colors whitespace-nowrap ${activeTab === tab.key ? 'border-emerald-500 text-emerald-600' : 'border-transparent text-slate-500 hover:text-slate-700'}`}>
   322→              {tab.icon} {tab.label}
   323→              {tab.key === 'messages' && unreadCount > 0 && <span className="ml-1 px-1.5 py-0.5 text-xs bg-red-500 text-white rounded-full">{unreadCount}</span>}
   324→            </button>
   325→          ))}
   326→        </div>
   327→
   328→        {/* Content */}
   329→        <div className="flex-1 overflow-y-auto custom-scrollbar p-6">
   330→
   331→          {/* SETTINGS */}
   332→          {activeTab === 'settings' && (
   333→            <div className="space-y-6 max-w-2xl">
   334→              <h3 className="text-lg font-semibold text-slate-900">Site Settings</h3>
   335→              <div className="space-y-4">
   336→                <div><Label>Site Name</Label><Input value={settings.siteName || ''} onChange={(e) => setSettings({ ...settings, siteName: e.target.value })} className="mt-1.5" /></div>
   337→                <div><Label>Tagline</Label><Input value={settings.tagline || ''} onChange={(e) => setSettings({ ...settings, tagline: e.target.value })} className="mt-1.5" /></div>
   338→                <div><Label>WhatsApp Number</Label><Input value={settings.whatsappNumber || ''} onChange={(e) => setSettings({ ...settings, whatsappNumber: e.target.value })} className="mt-1.5" /></div>
   339→                <div><Label>Contact Email</Label><Input value={settings.contactEmail || ''} onChange={(e) => setSettings({ ...settings, contactEmail: e.target.value })} className="mt-1.5" /></div>
   340→                <div><Label>About Title</Label><Input value={settings.aboutTitle || ''} onChange={(e) => setSettings({ ...settings, aboutTitle: e.target.value })} className="mt-1.5" /></div>
   341→                <div><Label>About Description</Label><Textarea value={settings.aboutDescription || ''} onChange={(e) => setSettings({ ...settings, aboutDescription: e.target.value })} className="mt-1.5" rows={4} /></div>
   342→                <div>
   343→                  <div className="flex justify-between items-center mb-3"><Label>Statistics</Label><Button size="sm" variant="outline" onClick={() => setAboutStats([...aboutStats, { label: '', value: '' }])} className="text-xs"><Plus size={14} className="mr-1" />Add</Button></div>
